Install External Dependencies
Setup the Python Packaging Virtual Environment

\lstset{caption=NFQL Parser on Debian/Ubuntu,
				tabsize=2, language=bash, numbers=left,stepnumber=1,
        basicstyle=\tiny\ttfamily, numberstyle=\ttfamily\color{gray},
        keywordstyle=\color{blue}, frame=shadowbox,
        rulesepcolor=\color{black}, label=lst:nfql-parser-ubuntu,
        aboveskip=20pt, captionpos=b, upquote=true}
\begin{lstlisting}
$ sudo apt-get install libhdf5-serial-dev
$ sudo apt-get install liblzo2-dev

$ sudo apt-get install python-pip
$ sudo pip install pip --upgrade
$ sudo pip install virtualenv
$ sudo pip install virtualenvwrapper
\end{lstlisting}

Create a Virtual Environment

\begin{lstlisting}
[parser] $ mkvirtualenv parser
\end{lstlisting}

Install Python Dependencies

\begin{lstlisting}
(parser)
[parser] $ make
\end{lstlisting}

List the Installed Dependencies

\begin{lstlisting}
(parser)
[parser] $ pip freeze
\end{lstlisting}

Cleanup:

Remove the build files

\begin{lstlisting}
(parser)
[parser] $ make clean
\end{lstlisting}

Deactivate the Virtual Environment

\begin{lstlisting}
(parser)
[parser] $ deactivate
\end{lstlisting}

Destroy the Virtual Environment

\begin{lstlisting}
[parser] $ rmvirtualenv parser  
\end{lstlisting}
